Line Graph

var data = [ [ 76, 70, 23, 41, 86, 59, 85, 57 ] ],
    options = {
        duration: 1000,
        padding: 20,
        normal_range: 0,
        line_width: 10,
        smooth_graph: true,
        peaks_and_valleys: true,
        bevel: true,
        dividers: [ { count: 25, line_width: 0.5 }, { count: 5, line_width: 1.5 } ],
        fill: [ 'rgba(0, 204, 255, 0.25)', 'rgba(0, 204, 255, 0)' ],
        fill: [
            'rgba(' + [ 0xff, 0xcc, 0, 0.25 ] + ')',
            'rgba(' + [ 0, 0xcc, 0xff, 0.25 ] + ')'
        ],
        color: [ '#fc0', '#0cf' ]
    };
$('#container').bobbograph( data, options );
